[해당하는 필드에 적용된 갯수를 알아볼 수 있음_COUNT]

:: 모든 필드 확인하기
SELECT * FROM stdscore;


:: "kor 필드" 에 작성되어 있는 데이터값 세어주기!
SELECT COUNT(kor) FROM stdscore;

:: 모든 필드에 작성되어 있는 데이터 값 세어주기!
SELECT COUNT(*) FROM stdscore;

	공간 0은 필수로 만들어진다!

	어떤값을 기준으로 세어줄때 "0" 은 매우 중요한 역할을 담당한다!

[조건절을 기준으로 출력받기]

: SELECT kor FROM stdscore
: FROM을 기준으로 앞쪽은 필드, 뒤쪽은 조건을 물어봄!

1) stdscore 테이블에서 이름이 red이고, 영어점수가 99점인 , kor 필드 호출받기

SELECT kor FROM stdscore WHERE name='red' AND eng='99';

[기본구문]
	SELECT 필드1, 필드2, 필드3 FROM 테이블 명 WHERE 필드명1='대입값1' AND 필드명2='필드명2';

	+ FROM을 기준으로 앞에서 필드명을 호출할 때는 (콤마) 사용
	+ FROM을 기준으로 뒤에서 조건을 만들때는 AND를 사용함!


[중요]
	"필드명" 을 호출할 때는 FROM 앞쪽이든, 뒷쪽이든 "필드명"은 절대 더블이나 싱글을 사용하지 않는다!
	"대입되는 값" 은 무조건 더블이나 싱글을 사용하여 호출한다!

	:: 숫자, 문자, 변수까지 대입되는 값이라면 전부 더블이나 싱글 사용!

	[매우중요, 참고]
		
		+ 필드명 = '문자';
		+ 필드명 = '숫자';
		+ 필드명 = '$name'; 
	
	:: 위에 처럼 필드명과 대입되는 값을 구분하기 위하여 대입되는 어떤 값도 전부 더블이나 싱글을 사용한다!
	:: 예약어는 대입되는 값이어도 더블이나 싱글을 사용하지 않는다!
