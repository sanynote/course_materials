(no int not null auto_increment primary key, fname varchar(30) not null, fyear varchar(30) not null, fprice int not null,fnum int not null)

12) comp 테이블 만들기

	+ no name, cpu, ram, mainboard, year, price 제작하기
	+ no 필드는 자동증가 설정하기, 유일한 키 설정하기
	+ price만 int 설정 나머지는 문자열 셋팅 (40)
	+ not null

	(문자열 셋팅하기)


CREATE TABLE comp
(no int not null auto_increment primary key, name varchar(40) not null, cpu varchar(40) not null, ram varchar(40) not null, mainboard varchar(40) not null, year varchar(40) not null, price int not null)
default charset=utf8;
